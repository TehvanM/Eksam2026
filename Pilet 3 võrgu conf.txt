RUUTER R1

enable
configure terminal
hostname R1
username admin privilege 15 secret Passw0rd!
enable secret Passw0rd!
ip domain-name cisco.local
crypto key generate rsa
1024
ip ssh version 2
line console 0
login local
password Passw0rd!
exit
line vty 0 4
login local
transport input ssh
password Passw0rd!
exit
banner motd #Autoriseerimata sisselogimine on keelatud!#
interface GigabitEthernet0/0
ip address 192.168.5.1 255.255.255.128
no shutdown
exit
interface Serial0/0/0
ip address 201.10.23.1 255.255.255.252
clock rate 64000
no shutdown
exit
interface Serial0/0/1
ip address 201.10.23.5 255.255.255.252
clock rate 64000
no shutdown
exit
router ospf 1
network 192.168.5.0 0.0.0.127 area 0
network 201.10.23.0 0.0.0.3 area 0
network 201.10.23.4 0.0.0.3 area 0
passive-interface GigabitEthernet0/0
exit
ip dhcp excluded-address 192.168.5.1 192.168.5.10
ip dhcp pool LAN
network 192.168.5.0 255.255.255.128
default-router 192.168.5.1
dns-server 1.1.1.1
domain-name too.net
exit
end
write memory
